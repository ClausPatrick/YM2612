# YM2612
Using ATMEGA8 serially interfaced via Python script to control FM synth YM2612

 Port0 = ym_D6
 Port1 = ym_D7
 Port2 = ym_!IRQ
 Port3 = ym_/0M
 Port4 = ym_!CS
 Port5 = ym_!WR

 Port0 = LED_RED1
 Port1 = ym_A0
 Port2 = ym_A1
 Port3 = LED_RED2
 Port2 = ym_D0
 Port3 = ym_D1

 Port0 = RXD
 Port1 = TXD
 Port2 = INT0
 Port3 = INT1
 Port4 = ym_D2
 Port5 = ym_D3
 Port6 = ym_D4
 Port7 = ym_D5
